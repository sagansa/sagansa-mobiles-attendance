# Attendance UI Fixes Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Apply 6 UI/UX fixes to the Sagansa Attendance app: SafeArea, password input visibility, onboarding permissions screen, modal refactoring, leave-request calendar/button layout, and icon/splash asset consistency.

**Architecture:** All changes are frontend-only (Expo SDK 57 / React Native 0.86 / expo-router). New reusable components (`PasswordInput`, `BaseModal`) are introduced and existing modals refactored onto them. A new standalone onboarding route with a guarded redirect is added. Icons are regenerated from `logo.jpg` via the existing `generate-icons.py` pipeline.

**Tech Stack:** Expo SDK 57, React Native 0.86, expo-router, `react-native-safe-area-context` 5.7, `@expo/vector-icons` (MaterialIcons), `expo-location`, `expo-image-picker`, Pillow+numpy (icon generation).

**Spec:** `doc/specs/2026-07-12-attendance-ui-fixes-design.md`

**Working context:** The working tree contains uncommitted WIP (tenant handling, logo assets, etc.). This plan builds on top of it. The login/register SafeArea + keyboard fix is **already applied** — do not redo it.

**Verification approach:** This app has no automated test harness. Each task verifies via `npx tsc --noEmit` (known pre-existing errors in `_layout.tsx`, `IconSymbol.tsx`, `useThemeColor.ts` are unrelated — filter for the files you touched) plus manual visual verification.

```tsx
import { useState } from 'react';
import { StyleSheet, TextInput, TouchableOpacity, View, type TextInputProps } from 'react-native';
import { MaterialIcons } from '@expo/vector-icons';

/**
 * Input kata sandi dengan:
 *  - warna teks hitam eksplisit (#111827) agar titik/teks selalu terlihat di light mode,
 *  - tombol mata untuk toggle lihat/sembunyikan sandi.
 *
 * Styling konsisten dengan field input lain di app (border #E5E7EB, radius 12).
 * Menerima seluruh props TextInput sehingga drop-in menggantikan <TextInput secureTextEntry />.
 */
export function PasswordInput(props: TextInputProps) {
  const [isSecure, setIsSecure] = useState(true);

  return (
    <View style={styles.wrap}>
      <TextInput
        {...props}
        style={[styles.input, props.style]}
        secureTextEntry={isSecure}
      />
      <TouchableOpacity
        style={styles.toggle}
        onPress={() => setIsSecure(prev => !prev)}
        accessibilityRole="button"
        accessibilityLabel={isSecure ? 'Tampilkan kata sandi' : 'Sembunyikan kata sandi'}
        hitSlop={{ top: 8, bottom: 8, left: 8, right: 8 }}
      >
        <MaterialIcons
          name={isSecure ? 'visibility' : 'visibility-off'}
          size={22}
          color="#6B7280"
        />
      </TouchableOpacity>
    </View>
  );
}

const styles = StyleSheet.create({
  wrap: {
    position: 'relative',
    justifyContent: 'center',
  },
  input: {
    backgroundColor: '#FFFFFF',
    borderRadius: 12,
    borderWidth: 1,
    borderColor: '#E5E7EB',
    paddingHorizontal: 16,
    paddingRight: 48,
    paddingVertical: 14,
    fontSize: 16,
    color: '#111827',
  },
  toggle: {
    position: 'absolute',
    right: 12,
    top: 0,
    bottom: 0,
    justifyContent: 'center',
    alignItems: 'center',
    width: 36,
  },
});
```

```tsx
import type { ReactNode } from 'react';
import { Modal, Pressable, StyleSheet, View } from 'react-native';
import { useSafeAreaInsets } from 'react-native-safe-area-context';

export type BaseModalVariant = 'center' | 'top-right' | 'full-screen';

export type BaseModalProps = {
  visible: boolean;
  onClose?: () => void;
  /** Default 'fade'. */
  animationType?: 'none' | 'slide' | 'fade';
  /** Default 'center'. */
  variant?: BaseModalVariant;
  /** Default 'rgba(17,24,39,0.4)'. */
  overlayColor?: string;
  /** Default true. When true, pressing the backdrop calls onClose. */
  dismissOnBackdrop?: boolean;
  children: ReactNode;
};

/**
 * Modal dasar reusable. Menangani <Modal> RN + overlay + backdrop-to-close +
 * penempatan konten per variant. Konsumen cukup menyediakan children (card/menu).
 *
 * variant:
 *  - 'center'      → konten di tengah layar (dialog).
 *  - 'top-right'   → konten menempel kanan-atas (menu dropdown), memakai SafeArea atas.
 *  - 'full-screen' → tanpa padding, anak mengisi layar (sheet/list penuh).
 */
export function BaseModal({
  visible,
  onClose,
  animationType = 'fade',
  variant = 'center',
  overlayColor = 'rgba(17, 24, 39, 0.4)',
  dismissOnBackdrop = true,
  children,
}: BaseModalProps) {
  const insets = useSafeAreaInsets();

  const handleBackdropPress = dismissOnBackdrop ? onClose : undefined;

  let overlayStyle;
  if (variant === 'center') {
    overlayStyle = [styles.overlayCenter, { backgroundColor: overlayColor }];
  } else if (variant === 'top-right') {
    overlayStyle = [
      styles.overlayTopRight,
      { backgroundColor: overlayColor, paddingTop: insets.top + 56 },
    ];
  } else {
    overlayStyle = [styles.overlayFullScreen, { backgroundColor: overlayColor }];
  }

  return (
    <Modal
      visible={visible}
      transparent
      animationType={animationType}
      onRequestClose={onClose}
    >
      <View style={overlayStyle}>
        <Pressable
          style={styles.backdrop}
          onPress={handleBackdropPress}
        />
        {children}
      </View>
    </Modal>
  );
}

const styles = StyleSheet.create({
  backdrop: {
    position: 'absolute',
    top: 0,
    left: 0,
    right: 0,
    bottom: 0,
  },
  overlayCenter: {
    flex: 1,
    alignItems: 'center',
    justifyContent: 'center',
    padding: 24,
  },
  overlayTopRight: {
    flex: 1,
    alignItems: 'flex-end',
    justifyContent: 'flex-start',
    paddingRight: 16,
  },
  overlayFullScreen: {
    flex: 1,
  },
});
```

```tsx
import { useCallback, useEffect, useState } from 'react';
import AsyncStorage from '@react-native-async-storage/async-storage';

const ONBOARDING_KEY = '@presence/onboarding-done';

/**
 * Membaca & menandai status onboarding (first-run permission flow).
 * Flag dipersist di AsyncStorage agar hanya muncul sekali per install.
 */
export function useOnboardingStatus() {
  const [isDone, setIsDone] = useState<boolean | null>(null);

  useEffect(() => {
    let mounted = true;
    void (async () => {
      const value = await AsyncStorage.getItem(ONBOARDING_KEY);
      if (mounted) {
        setIsDone(value === '1');
      }
    })();
    return () => { mounted = false; };
  }, []);

  const markDone = useCallback(async () => {
    await AsyncStorage.setItem(ONBOARDING_KEY, '1');
    setIsDone(true);
  }, []);

  return { isOnboardingDone: isDone, markOnboardingDone: markDone };
}
```
